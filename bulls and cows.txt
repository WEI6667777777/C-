#include<stdio.h>
#include<stdlib.h>
#include<time.h>
int main()
{
	srand(time(NULL));
	int num = rand() % 100 + 1;
	int count = 1;
	while (1)
	{
		
		int guess;
		printf("请输入你要猜的数字\n");
		scanf_s("%d", &guess);
		if (num < guess)
		{
			printf("大了\n");
			count=count+1;
		}
		else if (num > guess)
		{
			printf("小了\n");
			count = count + 1;
		}
		else
		{
			printf("恭喜你，猜中了\n");
			printf("你总共猜了%d次", count);
			break;
		}

	}
	return 0;
}
